# hasher
 Hash cracker with auto detect hash
 
# List hash supported
[01] md4\n
[02] md5\n
[03] sha1\n
[04] sha224\n
[05] sha256\n
[06] sha384\n
[07] sha512\n
[08] ripemd160\n
[09] whirlpool\n

# how to install

$ pkg update upgrade -y
$ pkg install python2 git
$ git clone https://github.com/ciku370/hasher
$ cd hasher
$ python2 hash.py
