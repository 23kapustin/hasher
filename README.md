# hasher
 Hash cracker with auto detect hash
 
# List hash supported
[01] md4
[02] md5
[03] sha1
[04] sha224
[05] sha256
[06] sha384
[07] sha512
[08] ripemd160
[09] whirlpool

# how to install

$ pkg update upgrade -y
$ pkg install python2 git
$ git clone https://github.com/ciku370/hasher
$ cd hasher
$ python2 hash.py
